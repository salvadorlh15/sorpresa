    <script>
        function showResponse(message) {
            document.getElementById('response').innerText = message;
            document.getElementById('response').classList.remove('hidden');
            confetti();
        }

        function confetti() {
            const confettiCount = 100;
            const confettiColors = ["#ff4081", "#ffeb3b", "#ff80ab", "#d81b60"];

            for (let i = 0; i < confettiCount; i++) {
                let confetto = document.createElement("div");
                confetto.style.position = "fixed";
                confetto.style.top = `${Math.random() * 100}%`;
                confetto.style.left = `${Math.random() * 100}%`;
                confetto.style.width = "10px";
                confetto.style.height = "10px";
                confetto.style.backgroundColor = confettiColors[Math.floor(Math.random() * confettiColors.length)];
                confetto.style.opacity = Math.random();
                confetto.style.transform = `rotate(${Math.random() * 360}deg)`;
                confetto.style.transition = "all 3s ease-out";
                document.body.appendChild(confetto);
                
                setTimeout(() => {
                    confetto.style.top = "100%";
                    confetto.style.opacity = "0";
                }, 100);
                
                setTimeout(() => {
                    confetto.remove();
                }, 3000);
            }
        }
    </script>
